def get_cluster_greedy_match(activation,vectors,n_clusters):
from sklearn.metrics.pairwise import euclidean_distances
num_vectors = vectors.shape[0]
cluster_size = num_vectors // n_clusters
dist = euclidean_distances(vectors).astype(np.float32)
assigned = np.zeros(num_vectors, dtype=bool)
cluster_list = []
with tqdm(total=n_clusters, desc="Clustering") as pbar:
while not assigned.all():
unassigned = np.where(~assigned)[0]
if len(unassigned) <= cluster_size:
act = activation[unassigned]

anchor_idx = unassigned[max_idx_in_cluster].item()
cluster_list.append({
"anchor": anchor_idx,
"neuron": unassigned.tolist(),
})
pbar.update(1)
break
sub_dist = dist[np.ix_(unassigned, unassigned)]
np.fill_diagonal(sub_dist, np.inf)
nearest = sub_dist.min(axis=1)
seed = unassigned[nearest.argmin()]
seed_dists = dist[seed, unassigned]
k = cluster_size # 包含 seed 自己
nearest_local = np.argsort(seed_dists)[:k]
members = unassigned[nearest_local]
act = activation[members]
max_idx_in_cluster = torch.argmax(act)
anchor_idx = members[max_idx_in_cluster].item()
cluster_list.append({
"anchor": anchor_idx,
"neuron": members.tolist(),
})
assigned[members] = True
pbar.update(1)
return cluster_list

def _make_grad_hook_text(orig_up, orig_gate, split_idx):
"""Create a backward hook that transfers proxy gradient to original weight parameters."""
def hook(grad):
up_grad = grad[:, :split_idx]
gate_grad = grad[:, split_idx:]
if orig_up.grad is not None:
orig_up.grad = orig_up.grad + up_grad
else:
orig_up.grad = up_grad.clone()
if orig_gate.grad is not None:
orig_gate.grad = orig_gate.grad + gate_grad
else:
orig_gate.grad = gate_grad.clone()
return hook
